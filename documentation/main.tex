\begin{algorithm}[H]
\SetAlgoLined

 //K - liczba określająca ile najlepszych par uwzględniamy\;

 //uniquePairs - określa czy pary mogą się powtarzać\;

 \If{czyLiść(dane)}{
  stwórzLiść();
 }
 
 pary = new List$<>$()\;
 
 \For{$i \gets 1$ to atrybuty.length$}{
  atrybut1 = atrybuty[i]\;
  \For{$j \gets i+1$ to atrybuty.length$}{
   atrybut2 = atrybuty[j]\;
   współczynnik = średnie[atrybut1] / średnie[atrybut2]\;
   int suma1, suma2 = 0\;
   \For{$x \gets $ to dane.length$}{ 
    \If{dane[x].value(artybut1) $>$ współczynnik * dane[x].value(artybut2)}{
      \eIf{dane[x].classValue()}{
       suma1++;
      }{
       suma2++;
      }
    }
   }
   p1 = suma1 / dane[klasa1].length\;
   p2 = suma2 / dane[klasa2].length\;
   delta = Math.abs(p1 - p2)\;
   pary.add(delta, atrybut1, atrybut2, współczynnik)\;
  }
 }
 
 pary.sort(); // Sortowanie po delcie malejąco
 
 najlepszePary = pary.wybierzNajlepsze(K, uniquePairs)\;
 
 podzieloneDane[] = podzielDane(dane, najlepszePary)\;
 
 średnie[] = średnieDlaAtrybutów(podzieloneDane)\;
 
 \For{$d \gets 1$ to podzieloneDane.length$}{
  buildTree(podzieloneDane[i], atrybuty, średnie, depth+1)\;
 }
 
 \caption{void buildTree(Instances dane, int[] atrybuty, double[] średnie, int depth)}
\end{algorithm}
